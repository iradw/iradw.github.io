# Promise分析及源码

# Promise是什么？

  • Promise是一个构造函数
  • Promise可以封装一个异步操作，并获取器结果

# Promise的状态改变

Promise对象共有3种状态，初识状态为Pending。Promise对象只能从Pending变成resolved状态，或者从Pending状态变成rejected状态，并且一个Promise对象只能改变一次状态。

# 为什么使用Promise

  • Promise可以链式调用，解决回调地狱的问题，如下

          function foo1(value, callback) {
          console.log('foo1'+value)
          return callback(value)
      }
      function foo2(value, callback) {
          console.log('foo2'+value)
          return callback(value)
      }
      function foo3(value, callback) {
          console.log('foo3'+value)
          return callback(value)
      }

      const p = foo1('value1', function() {
          return foo2('value2', function() {
              return foo3('value3', function() {
                  return 'aaa'
              })
          })
      })
      console.log(p)    //aaa

# 分析源码前要了解的关键问题

# .then()具体执行哪个回调函数由其所属的Promise对象决定

  const p = new Promise( (resolve, reject) => {
      resolve(1)
  })
  p.then(
      res => {
          console.log('resolve1 '+res)
          //1. 如果所属的Promise对象返回一个非Promise的值 则then()返回的Promise变为resolved状态， res为返回值
          // return 2         //输出resolve1 1    resolve2 2
          //2. 如果返回一个Promise对象，那下面的then()正常触发，由返回的Promise决定
          // return Promise.reject(3) //输出resolve1 1    reject 3
          //3. 如果抛出一个异常 则当前Promise对象变成rejected状态，err为Error对象
          // throw new Error('这里抛出错误')  // 输出resolve1 1    reject Error对象
          // 如果没有返回值 输出 resolve1 1  resolve2 undefined
      },
      err => {
          console.log('reject1 '+err)
      }
  ).then(
      res => {
          console.log('resolve2 '+res)
      },
      err => {
          console.log('reject2 '+err)
      }
  )

# 实现Promise内部结构

  const PENDING = 'pending'
  const FULFILLED = 'fullfilled'
  const REJECTED = 'reject'
  class MyPromise{
    constructor(executor) {}
    // 将Promise的状态改为fulfilled
    resolve(res) {}


    //将Promise的状态改为rejected
    reject(err) {}

    //指定promise成功时和失败时的回调
    then(onFulfilled, onRejected) {}

    catch(onRejected) {}

    static all(promises) {}

    static race(promises) {}
  }

# 构造函数constructor

  constructor(executor) {
      this.status = PENDING    // 初始状态pengding
      this.data = undefined    //promise携带的数据
      this.callbackQueue = [] // {onFulfilled: xxx, onRejected: xxxx} 也就是.then()里面的方法
      try{
         // 执行executor, executor执行时用户会改变Promise对象的状态
          executor(this.resolve.bind(this), this.reject.bind(this)) 
      }catch (error) {    //如果发生异常 捕获并把状态变为rejected，因此执行reject
          this.reject(error)  
      }   
  }

# resolve和reject

  resolve(res) {
      if (this.status !== PENDING){    // 每个Promise对象只能改变一次状态 也就是只能从pending到另外两种状态
          return
      }
      this.status = FULFILLED //变更状态
      this.data = res
      //onFulfilled
      if (this.callbackQueue.length > 0) {    //异步执行方法队列中的方法 并传入Promise对象中的参数
          this.callbackQueue.forEach( callbackObj => {
              setTimeout(() => {              // 通过setTimeout可以实现异步
                  callbackObj.onFulfilled(res)
              })
          })
      }
  }

  reject(err) {
      if (this.status !== PENDING) 
          return
      this.status = REJECTED
      this.data = err
      if (this.callbackQueue.length > 0) {
          this.callbackQueue.forEach( callbackObj => {
              setTimeout(() => {
                  callbackObj.onRejected(err)
              })
          })
      }
  }

# then方法

then是最复杂的方法，也是Promise的核心

  then(onFulfilled, onRejected) {
      const _this = this
      //保证onFulfilled和onRejected是函数
      onFulfilled = type.isFunction(onFulfilled) ? onFulfilled : value => value
      onRejected = type.isFunction(onRejected) ? onRejected : error => {throw error}

      //then()方法返回一个MyPromise对象，这个Promise是什么状态，data是什么， 由onResolved和onRejected决定
      return new MyPromise( (resolve, reject) => {
          //如果then方法执行时，Promise还是pending状态，只需要把onFulfilled和ronRejected方法放到执行队列中
          if (_this.status === PENDING) {
              _this.callbackQueue.push({
                  onFulfilled: function(){
                      handler(onFulfilled)
                  },
                  onRejected: function() {
                      handler(onRejected)
                  }
              })
          }
          // 如果then()方法执行时，promise对象已经时fulfilled状态 就要执行onFulfilled方法了
          else if (_this.status === FULFILLED) {
              setTimeout(()=>{
                  handler(onFulfilled)
              })
          }
          // 如果then()方法执行时，promise对象已经时fulfilled状态 就要执行onRejected方法了
          else {
              setTimeout(()=>{
                  handler(onRejected)
              })
              
          }
          
          //根据上一个Promise的状态及结果处理当前返回的Promise, func为onResolved或onRejected
          function handler(func) {
              try {
                  const result = func(_this.data)
                  /*
                  下面问号后面可能不太好理解 当前Promise的状态依赖 
                  于result这个promise的状态，
                  result是什么状态我们不知道，取决于用户的调用，如果result成功了，
                  必然会去执行result.then()中的第一个参数，把当前promise 
                  的改变状态的resolve函数作为result成功的回调，
                  不就改变当前Promise对象的状态为resolved吗？和下面的写 
                  法相同，只不过下面的写法多了一层没用的函数
                      result.then(
                          res => resolve(res),
                          err => reject(err)
                      )
              */
                  result instanceof MyPromise ? result.then(resolve, reject) : resolve(result)
              } catch (error) {
                  reject(error)
              }
          }
      })
  }

也许你会有一个疑问，onFulfilled和onRejected不是已经在执行resolve或reject时调用过了吗？其实不会，我们上面说过，then方法可以在任何时候执行，大多数情况下，then是后执行的，Promise中的executor先执行, 也就是说执行resolve和reject时, callbackQueue中还没有方法，只改变了Promise对象的状态，并且保存了data， 当then执行时，判断Promise对象的状态，如果是"fulfilled"就调用onResolved, 如果是"rejected"就调用onRejected 。

还有情况是then先执行了，Promise对象的状态还没改变，也就是"pending",这时需要把onResolved和onRejected方法放到callbackQueue中，以便Promise对象状态改变时调用。 其中，做类型判断的type见文章判断数据类型的问题

# Promise.all

返回一个新的MyPromise对象，当传入的所有MyPromise对象都执行成功，返回的MyPromise对象就是成功的，且携带的data是一个数组，里面按顺序存放这参数中的promise对象执行成功时的data

  static all(promises) {
      const length = promises.length  
      const values = new Array(length)    //提前指定数组的长度
      let counter = 0     //用来记录 promise执行完成的个数
      return new MyPromise( (resolve, reject) => {
          promises.forEach( (promise, index) => {
              promise.then(               // 获取每个promise的data
                  res => {
                      values[index] = res
                      counter++           
                      if (counter === length) {   //所有promise对象的值都拿到了
                          resolve(values)
                      }
                  },
                  err => {                //有一个promise对象出错 返回的promise对象变为rejected状态
                      reject(err)
                  }
              )
          })
      })
      
  }

# Promise.race

返回一个Promise对象，状态和data和最先执行完成的promise对象一致

  static race(promises) {
      return new MyPromise( (resolve, reject) => {
          promises.forEach( promise => {
              promise.then(
                  res => {
                      resolve(res)
                  },
                  err => {
                      reject(err)
                  }
              )
          })
      })
  }
