gitlab+jenkins的持续集成

DivYang 2020-08-13 CI/CD

# 配置jenkins

  1. 在jenkins中新建流水线

  2. 填写gitlab项目的地址，并勾选‘当gitlab收到改变代码的push'时的hook

  3. 配置触发的分支，支持正则并生成Token

# 配置gitlab

  1. 配置gitlab的webhooks,填写jenkins项目地址以及刚刚生成的Token

  2. 配置好以后可以在webhooks下方看到Project Hooks，点击Test => push event

    如果在jenkins页面中看到构建记录，则配置成功。

# 配置项目

添加Dockerfile和jenkinsfile

Dockerfile

FROM docker.nroad.com.cn/node:10 as frontend-builder

WORKDIR /frontend-build

COPY . /frontend-build

RUN set -x \
    && npm config set registry https://nexus.nroad.com.cn/repository/npm-public/ \
    && npm config set _auth amVua2luczpqZW5raW5z \
    && npm config set email zhoupen9@sina.cn \
    && npm install \
    && npm run build

COPY package.json ./dist/

RUN npm publish dist

FROM docker.nroad.com.cn/nginx

WORKDIR /

COPY --from=frontend-builder /frontend-build/dist /usr/share/nginx/html

Jenkinsfile

def label = "w-pw-ja-${UUID.randomUUID().toString()}"

podTemplate(
    label: label,
    containers: [
        containerTemplate(name: 'docker', image: 'docker.nroad.com.cn/docker:18', command: 'cat', ttyEnabled: true)
    ]
) {
    node(label) {
        stage('checkout') {
            checkout scm

            env.APP_VERSION = sh(returnStdout: true,
                script: "cat package.json | grep version | head -1 | awk -F: '{ print \$2 }' | sed 's/[\",]//g' | tr -d '[[:space:]]'")
            env.WORKSPACE = pwd()
            env.DOCKER_HOST = "tcp://dm0.nroad.com.cn:2375"
            env.REGISTRY = "docker.nroad.com.cn/soc"
            env.IMAGE = "large-screen"
        }

        stage('build') {
            container('docker') {
                sh "docker --host ${env.DOCKER_HOST} build \
                           --file ${env.WORKSPACE}/Dockerfile \
                           --tag ${env.REGISTRY}/${env.IMAGE}:${env.APP_VERSION} \
                            ${env.WORKSPACE}"
                sh "docker --host ${env.DOCKER_HOST} push ${env.REGISTRY}/${env.IMAGE}:${env.APP_VERSION}"
            }
        }
