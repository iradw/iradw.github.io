# 函数柯里化

# 柯里化的作用

柯里化可以实现这样一个功能

curryAdd(1)(2, 3)(4)  // 10

add是柯里化之后的函数，柯里化之前就是一个普通的求和函数

function add(n1, n2, n3, n4){return n1 + n2 + n3 + n4}

也就是说普通的add函数经过处理，变成了curryAdd，下面介绍这个处理函数currying。

# 存储传入的参数

柯里化函数的结构肯定如下, 以刚开始的例子做分析, fn为上述的add，currying的返回值为上述的curryAdd

function currying(fn) {
    
    return function (){    
            //在这里判断何时调用fn
    }
}

function currying(fn) {

function currying(fn) {
    let params = []    //把currAdd的参入1, 2, 3, 4放到这个数组
    return function (){
            params = [...params, ...arguments]  // 把arguments放到params中

}

# 判断fn何时触发原函数fn

函数有一个length属性，表示函数的形参个数，当params数组存储了fn.length个参数的时候，就可以触发原函数了，如果params里面的参数还不够，那就把当前函数在返回。

function currying(fn){
    const limit = fn.length
    let params = [] 
    return function (){
        params = [...params, ...arguments]  
        if (params.length >= limit){ // 参数数量到limit 可以执行fn了，返回结果
            return fn.call(null, params)
        }else {
            return arguments.callee    //返回当前函数  也可以给当前函数命名，直接放回函数名
        }
    }
}

# 指定调用次数

如果有一个未知形参个数的万能函数，如

function add(...args) {
    return args.reduce( (sum, current) => sum+current )
}

2
3

可以把柯里化函数的调用次数指定

function currying(fn, length){
    const limit = length || fn.length
    let params = [] 
    return function (){
            params = [...params, ...arguments]  
            if (params.length >= limit){ // 参数数量到limit 可以执行fn了，返回结果
                return fn.call(null, params)
            }else {
                return arguments.callee    //返回当前函数  也可以给当前函数命名，直接放回函数名
            }
        }
}

const curryAdd5 = currying(add, 5)
const curryAdd3 = currying(add, 3)
console.log(curryAdd5(1)(2)(3,4)(5))      //15
console.log( curryAdd3(3)(3, 3) )       //9

# 柯里化应用

function myEventListener(element, eventName) {
    return function(func) {    // func为事件处理函数
        element.addEventListener(eventName, function() {
            func.call(this, ...arguments)
        }, false)
    }
}
const bodyClickListener = myEventListener(document.body, 'click')
bodyClickListener(function (e) {    //不必重复的写addEventListener这个繁琐的api了
    console.log('1')
})
bodyClickListener(function (e) {
    console.log('2')
})
